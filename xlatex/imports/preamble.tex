\newenvironment{formula}[1]{
  \stepcounter{formula}
  \begin{tcolorbox}[
    standard jigsaw, % Allows opacity
    colframe={\IfDarkModeTF{white}{black}},
    boxrule=1px,
    colback=black,
    opacityback=0,
    sharp corners,

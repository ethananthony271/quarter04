\documentclass{standalone}
\input{../../../xlatex/imports/preamble}

\title{Right Quadrant Graph}
\date{}

\begin{document}
\begin{tikzpicture}
  % Right Quadrants Graph
  \def \xPosLen {5}
  \def \yPosLen {5}
  \def \yNegLen {5}
  \def \xOrigin {0}
  \def \yOrigin {0}
  \pgfmathsetmacro \xpf {\xOrigin + \xPosLen}
  \pgfmathsetmacro \ypf {\yOrigin + \yPosLen}
  \pgfmathsetmacro \ynf {\yOrigin - \yNegLen}
  \draw[graphLine] (\xOrigin,\yOrigin) -- (\xpf,\yOrigin) ;
  \draw[graphLine] (\xOrigin,\yOrigin) -- (\xOrigin,\ypf) ;
  \draw[graphLine] (\xOrigin,\yOrigin) -- (\xOrigin,\ynf) ;
\end{tikzpicture}
\end{document}
